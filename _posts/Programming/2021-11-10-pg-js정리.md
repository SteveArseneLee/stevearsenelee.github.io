## 함수
```js
function test(a){
  console.log(a)
}
test(10)
```

인자 두개
```js
function sum(a, b){
  return a+b;
}
var c = sum(10,20);
console.log(c);
```

입력과 출력 둘 다 없는 경우
```js
function sum(){
  console.log("sum 함수를 호출했나요");
}
var c = sum();
```

var을 이용한 익명함수
```js
var sum = function(){
}
```

this 사용
```js
function Car(a,b,c){
  this.name = a;
  this.color = b;
  var move = c;
}

var a = new Car("현대", "노랑")
console.log(a.name, a.color);
console.log(a.move);

var b = new Car("기아", "파랑")
console.log(b.name, b.color);
console.log(b.move);
```

## 프로토타입
```js
function Car(a,b){
  this.name = a;
  this.color = b;
}
Car.prototype.move = function() {
  console.log(this.name + "차이고 " + this.color + "색입니다.");
}

var a = new Car("현대", "노란")
a.move()
var b = new Car("기아", "파랑")
b.move()
```
